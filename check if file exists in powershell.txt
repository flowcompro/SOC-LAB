Check if a file exists
$fileToCheck = "C:\tmp\test.txt"
if (Test-Path $fileToCheck -PathType leaf)
{
    #do some stuff
}
Delete file if it exists
$fileToCheck = "C:\tmp\test.txt"
if (Test-Path $fileToCheck -PathType leaf)
{
    Remove-Item $fileToCheck
}
An even shorter method….

$fileToCheck = "c:\tmp\test.txt"
Remove-Item $fileToCheck -ErrorAction Ignore
